<template>
  <div id="spell-page" ref="page">
    <rest-page v-show="showRest"></rest-page>
    <div id="spell-background">
      <div class="spell-top">
        <p>当前生词数：{{newWordCount}}&nbsp;&nbsp;&nbsp;回想次数：{{recallCount}}</p>
        <div class="timer-music">
          <tomato-timer></tomato-timer>
          <student-background-music></student-background-music>
        </div>
      </div>
      <div class="spell-content">
        <div class="spell-interpretation-and-pronunciation">
          <h1>{{curWord}}</h1>
          <div class="listen-button-group">
            <div class="pronunciation-css">{{curWordInfo.pronunciation}}</div>
            <button>
              <img
                class="listen-picture"
                src="../../assets/student_listen_true.png"
                alt="listening"
              />英
            </button>
            <button>
              <img
                class="listen-picture"

            </button>
          </div>
        </div>
        <key-board v-if="false"></key-board>
        <make-choice v-if="cardOne" @know="knowChoice" @donnotknow="noChoice" @kill="killChoice"></make-choice>
        <show-tips
          :wordTip="curWordInfo"
          v-if="cardTwo"
          @recallBtn="recall"
          @noRecallBtn="noRecall"
        ></show-tips>
        <recall-part
          v-if="cardThree"
          :showTwo="showBtn"
          :wordTip="curWordInfo"
          @nextWord="toNextWord"
          @recallWrong="recallWrongBtn"
          @recallRight="recallRightBtn"

<script>
import StudentBackgroundMusic from '../../components/StudentBackgroundMusic'
import TomatoTimer from '../../components/TomatoTimer'
import KeyBoard from '../../components/KeyBoard'
import MakeChoice from '../../components/MakeChoice'
import ShowTips from '../../components/ShowTips'
import RestPage from './RestPage'
import axios from 'axios'
import RecallPart from '../../components/RecallPart'
 
export default {
  name: 'Spell',
  components: {
    StudentBackgroundMusic,
    TomatoTimer,
    KeyBoard,
    MakeChoice,
    ShowTips,
    RestPage,
    RecallPart
  },
  data () {
    return {
      newWordCount: 3,
      recallCount: 0,
      showRest: false,
      curIndex: 0,
      curGroupWords: [],
      curWord: '',
      len: '',
      cardOne: true,
      cardTwo: false,
      cardThree: false,
      curWordInfo: '',
      showBtn: '',
      sequenceHandle: ''
    }
  },
  watch: {
    curIndex () {
      this.curWord = this.curGroupWords[this.curIndex].word
      this.curWordInfo = this.curGroupWords[this.curIndex]
    }
  },
  methods: {
    knowChoice () {
      this.sequenceHandle = this.sequenceHandle + '1'
      this.showBtn = true
      this.cardTwo = false
      this.cardOne = false
      this.cardThree = true
    },
    noChoice () {
      this.sequenceHandle = this.sequenceHandle + '0'
      this.cardOne = false
      this.cardTwo = true
    },
    killChoice () {
      const that = this
      that.sequenceHandle = that.sequenceHandle + '2'
      if (that.curIndex + 1 < that.len) {
        that.curIndex = that.curIndex + 1
        that.newWordCount = that.newWordCount - 1
      } else {
        that.$router.push({
          path: '/StudentMain',
          name: 'StudentMain',
          params: {
            studentId: that.cookies.get('id'),
            studentInfo: that.cookies.get('info')
          }
        })
      }
    },
    recall () {
      this.sequenceHandle = this.sequenceHandle + '1'
      this.showBtn = true
      this.cardTwo = false
      this.cardOne = false
      this.cardThree = true
    },
    noRecall () {
      this.sequenceHandle = this.sequenceHandle + '0'
      this.showBtn = false
      this.cardTwo = false
      this.cardOne = false
      this.cardThree = true
    },
    toNextWord () {
      this.sequenceHandle = this.sequenceHandle + '3'
      this.killChoice()
      this.cardTwo = false
      this.cardOne = true
      this.cardThree = false
    },
    recallWrongBtn () {
      this.sequenceHandle = this.sequenceHandle + '0'
      this.toNextWord()
    },
    recallRightBtn () {
      this.sequenceHandle = this.sequenceHandle + '1'
      this.toNextWord()
      this.newWordCount = this.newWordCount - 1
      this.recallCount = this.recallCount + 1
    },
    getScreenHeight: function () {
      const outter = this.$refs.page
      outter.style.height = window.innerHeight + 'px'
    },
    getData: function (unitName, categoty) {
      const that = this
      axios
        .get(that.baseUrl + '/words/getgroupwords', {
          params: {
            id: that.cookies.get('id'),
            unit: unitName,
            category_id: categoty
          }
        })
        .then(res => {
          that.curGroupWords = res.data.words
          that.curWord = res.data.words[0].word
          that.curWordInfo = res.data.words[0]
          that.len = res.data.words.length
        })
    }
  },
  mounted () {
    let unitName = this.$route.params.unitName
    let category = 1
    this.getData(unitName, category)
    this.getScreenHeight()
  }
}
</script>
