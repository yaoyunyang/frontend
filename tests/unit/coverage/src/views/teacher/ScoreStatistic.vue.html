<template>
  <div>
    <el-container>
      <teacher-header></teacher-header>
    </el-container>
    <el-container class="container">
      <el-aside width="200px" class="aside">
        <teacher-bar></teacher-bar>
      </el-aside>
      <div class="outer-protect">
        <Collapse class="outer-collapse">
          <Panel name="class-choose">
            统计说明
            <div slot="content" class="panel-diatance"></div>
          </Panel>
          <Panel name="student-choose">
            请选择班级
            <p slot="content" class="student-choose-content">
              <el-form>
                <el-form-item>
                  <el-select v-model="value" placeholder="请选择班级">
                    <el-option
                      v-for="item in classOptions"
                      :key="item.value"
                      :label="item.label"
                      :value="item.label"
                    ></el-option>
                  </el-select>
                  <el-button plain class="button-distanse" @click="statisticType = 'train'">训练量</el-button>
                  <el-button plain @click="statisticType = 'word'">词汇量</el-button>
                  <el-button plain @click="statisticType = 'time'">学习时间</el-button>
                  <el-button plain @click="statisticType = 'train_word'">训练&词汇</el-button>
                </el-form-item>
              </el-form>
              <ve-histogram :data="chartData" width="1000px" height="500px"></ve-histogram>
            </p>
          </Panel>
        </Collapse>
      </div>
    </el-container>
  </div>
</template>
 
<script>
import TeacherBar from '../../components/TeacherBar.vue'
import TeacherHeader from '../../components/TeacherHeader.vue'
 
const axios = require('axios')
export default {
  components: {
    TeacherBar,
    TeacherHeader
  },
  data () {
    return {
      classOptions: [],
      value: '',
      statisticType: '',
      chartData: {
        columns: [],
        rows: []
      }
    }
  },
  watch: {
    statisticType: function () {
      this.showStatistic()
    }
  },
  mounted: function () {
    this.getAllClass()
  },
  methods: {
    getAllClass () {
      const that = this
      axios.get(that.baseUrl + '/words/showallclass/').then(response => {
        let res = response.data
        if (res['code'] === 200) {
          that.classOptions = res['data']
        }
      })
    },
    showStatistic () {
      const that = this
      axios
        .get(that.baseUrl + '/words/showstatistic', {
          params: {
            classId: that.value,
            statisticType: that.statisticType
          }
        })
        .then(response => {
          let res = response.data
          if (res['code'] === 200) {
            that.chartData['columns'] = res['data']['columns']
            that.chartData['rows'] = res['data']['rows']
          }
        })
    }
  }
}
</script>
