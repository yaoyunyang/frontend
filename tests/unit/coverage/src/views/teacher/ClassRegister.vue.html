<script>
import TeacherHeader from '../../components/TeacherHeader.vue'
 
const axios = require('axios')
export default {
  name: 'ClassRegister',
  components: {
    TeacherHeader
  },
  data () {
    return {
      classes: [],
      value: '',
      students: [],
      studentsValue: '',
      types: [
        {
          typeValue: '选项一',
          typeLabel: '小学英语'
        },
        {
          typeValue: '选项二',
          typeLabel: '初中英语'
        },
        {
          typeValue: '选项三',
          typeLabel: '高中英语'
        },
        {
          typeValue: '选项四',
          typeLabel: '大学英语'
        },
        {
          typeValue: '选项五',
          typeLabel: '考试英语'
        },
        {
          typeValue: '选项六',
          typeLabel: '行业英语'
        },
        {
          typeValue: '选项七',
          typeLabel: '其他'
        }
      ],
      typeValue: '',
      courses: [],
      courseValue: '',
      coursesData: [],
      courseId: '',
      coursePwd: '',
      fuzzySearch: '',
      registerSuccessVisible: false,
      registerFailureVisible: false
    }
  },
  watch: {
    value: function () {
      this.studentsValue = ''
      this.getAllStudent()
    },
    typeValue: function () {
      this.courseValue = ''
      this.getAllCourse()
    },
    studentsValue: function () {
      this.getAllRegisterCourse()
    }
  },
  mounted: function () {
    this.getAllClass()
  },
  methods: {
    getAllClass () {
      const that = this
      axios.get(that.baseUrl + '/words/showallclass/').then(response => {
        let res = response.data
        if (res['code'] === 200) {
          that.classes = res['data']
        }
      })
    },
    getAllStudent () {
      const that = this
      axios
        .get(that.baseUrl + '/words/showallstudent/', {
          params: {
            classId: that.value
          }
        })
        .then(response => {
          let res = response.data
          if (res['code'] === 200) {
            that.students = res['data']
          }
        })
    },
    getAllCourse () {
      const that = this
      axios
        .get(that.baseUrl + '/words/getallcourse/', {
          params: {
            typeValue: that.typeValue
          }
        })
        .then(response => {
          let res = response.data
          if (res['code'] === 200) {
            that.courses = res['data']
          }
        })
    },
    checkoutCategory () {
      const that = this
      axios
        .get(that.baseUrl + '/words/checkoutcategory/', {
          params: {
            studentId: that.studentsValue,
            categoryId: that.courseId,
            password: that.coursePwd
          }
        })
        .then(response => {
          let res = response.data
          if (res['code'] === 200) {
            that.registerSuccessVisible = true
            that.getAllRegisterCourse()
          } else {
            that.registerFailureVisible = true
          }
        })
    },
    getAllRegisterCourse () {
      const that = this
      axios
        .get(that.baseUrl + '/words/getallregistercourse/', {
          params: {
            studentId: that.studentsValue
          }
        })
        .then(response => {
          let res = response.data
          if (res['code'] === 200) {
            that.coursesData = res['data']
          }
        })
    }
  }
}
</script>
